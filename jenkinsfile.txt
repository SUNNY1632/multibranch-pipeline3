 pipeline {
    agent any

    stages {
        stage('Identify and Deploy') {
            steps {
                script {
                    if (env.BRANCH_NAME == 'development') {
                        sh './deploy-development.sh'
                    } else {
                        sh 'echo "This pipeline is configured only for the development branch. Current branch: ${env.BRANCH_NAME}"'
                    }
                }
            }
        }
    }
}
